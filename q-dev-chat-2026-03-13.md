### 4. **Collapse Manager**

```python
class CollapseManager:
    """Manages collapse/expand state for any node."""
    
    def __init__(self):
        self.collapsed_nodes: Set[str] = set()
        self.node_groups: Dict[str, List[str]] = {}
    
    def collapse_node(self, node_id: str, children: List[str]):
        """Collapse a node and hide its children."""
        self.collapsed_nodes.add(node_id)
        self.node_groups[node_id] = children
    
    def expand_node(self, node_id: str):
        """Expand a node and show its children."""
        self.collapsed_nodes.discard(node_id)
    
    def is_collapsed(self, node_id: str) -> bool:
        """Check if node is collapsed."""
        return node_id in self.collapsed_nodes
    
    def get_visible_nodes(self, all_nodes: List[str]) -> List[str]:
        """Get list of nodes that should be visible."""
        visible = []
        for node in all_nodes:
            # Check if node is hidden by a collapsed parent
            if not self._is_hidden_by_parent(node):
                visible.append(node)
        return visible
```

### 5. **External Grouping System**

```python
class ExternalGrouper:
    """Groups external dependencies by package."""
    
    def create_groups(self, nodes: Dict[str, CodeNode]) -> Dict[str, List[str]]:
        """Create grouped nodes for external dependencies."""
        groups = {
            'stdlib': [],
            'third_party': [],
            'builtin': []
        }
        
        for node_id, node in nodes.items():
            if node.is_external:
                groups[node.package_type].append(node_id)
        
        return groups
    
    def create_group_nodes(self, groups: Dict[str, List[str]]) -> Dict[str, CodeNode]:
        """Create collapsed group nodes."""
        group_nodes = {}
        
        for group_type, members in groups.items():
            if members:
                group_id = f"__group__{group_type}"
                group_nodes[group_id] = CodeNode(
                    name=f"📦 {group_type.replace('_', ' ').title()} ({len(members)})",
                    node_type='external_group',
                    file_path='',
                    line_number=0,
                    is_external=True,
                    is_collapsed=True,
                    children=members
                )
        
        return group_nodes
```
